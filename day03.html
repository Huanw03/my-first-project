    • `; }).join(''); // 关键：转为字符串，避免页面出现逗号，数组默认用,连接 console.log('收藏列表HTML:', favoriteHtml); // 实战：渲染到页面 const favoriteList = document.createElement('ul'); favoriteList.className = 'favorite-list';为新的列表取一个类名 favoriteList.innerHTML = favoriteHtml; document.body.appendChild(favoriteList);标签中增加列表项 // 3. filter:筛选含“工具”的收藏项（返回符合条件的新数组）//筛选符合条件的元素 const toolFavorites = favorites.filter(item => { return item.includes('工具'); // 筛选条件：包含“工具” }); console.log('=== 工具类收藏项 ===', toolFavorites); // 输出：["思维导图工具", "截图工具"] // 实战：渲染筛选后的列表 const toolList = document.createElement('ul'); toolList.className = 'tool-favorite-list'; toolList.innerHTML = toolFavorites.map(item => `
